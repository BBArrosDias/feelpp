macro(feel_append_src DIRNAME FILES)
  foreach(FILE ${FILES})
    list(APPEND LIST ${DIRNAME}/${FILE})
  endforeach(FILE)
  set(FEEL_SRCS ${FEEL_SRCS};${LIST} PARENT_SCOPE)
  set(FEEL_DIRS ${FEEL_DIRS};${DIRNAME} PARENT_SCOPE)
endmacro(feel_append_src)
