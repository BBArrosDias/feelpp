add_custom_target(gentoto ALL COMMENT "Generate Feel Performance Stokes files"   WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR} )
macro(OVERWITE_IF_DIFFERENT filename var)
  file(WRITE ${CMAKE_CURRENT_BINARY_DIR}/copy_${filename} ${var})
  if ( NOT EXISTS ${CMAKE_CURRENT_BINARY_DIR}/${filename} )
    add_custom_command(TARGET gentoto COMMAND ${CMAKE_COMMAND} -E copy
      ${CMAKE_CURRENT_BINARY_DIR}/copy_${filename} ${CMAKE_CURRENT_BINARY_DIR}/${filename}
      WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR})
  else()
    add_custom_command(TARGET gentoto COMMAND ${CMAKE_COMMAND} -E copy_if_different
      ${CMAKE_CURRENT_BINARY_DIR}/copy_${filename} ${CMAKE_CURRENT_BINARY_DIR}/${filename}
      WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR})
  endif()
endmacro()
