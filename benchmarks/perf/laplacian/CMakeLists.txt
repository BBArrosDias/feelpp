add_custom_target(gentoto ALL COMMENT "Generate Feel Performance Laplacian files"   WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR} )
macro(OVERWITE_IF_DIFFERENT filename var)
  file(WRITE ${CMAKE_CURRENT_BINARY_DIR}/copy_${filename} ${var})
  if ( NOT EXISTS ${CMAKE_CURRENT_BINARY_DIR}/${filename} )
    add_custom_command(TARGET gentoto COMMAND ${CMAKE_COMMAND} -E copy
      ${CMAKE_CURRENT_BINARY_DIR}/copy_${filename} ${CMAKE_CURRENT_BINARY_DIR}/${filename}
      WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR})
  else()
    add_custom_command(TARGET gentoto COMMAND ${CMAKE_COMMAND} -E copy_if_different
      ${CMAKE_CURRENT_BINARY_DIR}/copy_${filename} ${CMAKE_CURRENT_BINARY_DIR}/${filename}
      WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR})
  endif()
endmacro()

include_directories(${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_BINARY_DIR})

#foreach(N 2 3 4 5)
foreach(S "Simplex" "Hypercube")
foreach(D 2)
  foreach(N 2 3 4 5 )
    math (EXPR P '${N}-1')
    set( LAG "// this file is automatically generated
#include <laplacian.hpp>
namespace Feel {
  template class Laplacian<${D}, Lagrange<${N}, ${S}>\\;
}
")
    OVERWITE_IF_DIFFERENT("feel_perf_laplacian_${S}_${D}D_p${N}p${P}.cpp" "${LAG}")
    set(lagrange "${CMAKE_CURRENT_BINARY_DIR}/feel_perf_laplacian_${S}_${D}D_p${N}p${P}.cpp;${lagrange}")
  endforeach()
endforeach()
endforeach()


add_executable(feel_perf_laplacian  ${CMAKE_CURRENT_BINARY_DIR}/feel_perf_laplacian_Hypercube_2D_p2p1.cpp ${CMAKE_CURRENT_BINARY_DIR}/feel_perf_laplacian_Hypercube_cr1p0.cpp bench.cpp)
add_dependencies(feel_perf_laplacian gentoto)
target_link_libraries(feel_perf_laplacian ${FEEL_LIBRARIES})


configure_file(laplacian.cfg laplacian.cfg)
