cmake_minimum_required(VERSION 2.8)
find_package(HDF5 REQUIRED)
if (HDF5_FOUND)
    message (STATUS "HDF5 - Headers ${HDF5_INCLUDE_DIRS}")
    message (STATUS "HDF5 - Libraries ${HDF5_LIBRARIES}")
    include_directories (${HDF5_INCLUDE_DIRS})
endif ()
find_package(MPI REQUIRED)
if (MPI_FOUND)    
    message (STATUS "MPI - Headers ${MPI_INCLUDE_DIRS}")
    message (STATUS "MPI - Libraries ${MPI_LIBRARIES}")
    include_directories (${MPI_INCLUDE_PATH})
endif ()
include_directories (${Boost_INCLUDE_DIRS})
add_executable (exporterhdf5 exporterhdf5.cpp)
feelpp_add_application (exporterhdf5 SRCS exporterhdf5.cpp CFG exporterhdf5.cfg)
add_executable (hdf5io hdf5io.cpp)
add_executable (main main.cpp)
add_executable (mylaplacian mylaplacian.cpp)
link_libraries (feelpp ${FEELPP_LIBRARIES})
target_link_libraries (hdf5io feelpp ${MPI_LIBRARIES} ${HDF5_LIBRARIES})
target_link_libraries (exporterhdf5 feelpp ${MPI_LIBRARIES} ${HDF5_LIBRARIES} ${FEELPP_LIBRARIES} ${Boost_LIBRARIES})
target_link_libraries (mylaplacian feelpp ${MPI_LIBRARIES} ${HDF5_LIBRARIES} ${FEELPP_LIBRARIES} ${Boost_LIBRARIES})
target_link_libraries (main feelpp ${MPI_LIBRARIES} ${HDF5_LIBRARIES} ${FEELPP_LIBRARIES} ${Boost_LIBRARIES})
